53       @SuppressWarnings("deprecation") 
54       @Override 
55       public View getView(int position, View convertView, ViewGroup parent) { 
56            
57           ViewHolder holder = null; 
58            
59           if (convertView == null) { 
60               convertView = LayoutInflater.from(mInstance).inflate(R.layout.listview_item, null); 
61                
62               holder = new ViewHolder();  
63               holder.checkBox = (CheckBox) convertView.findViewById(R.id.check_box); 
64               holder.textView = (TextView) convertView.findViewById(R.id.app_name); 
65               holder.imageView = (ImageView) convertView.findViewById(R.id.app_logo); 
66                
67               convertView.setTag(holder); 
68           }else{ 
69               holder = (ViewHolder) convertView.getTag(); 
70           } 
71            
72           final PackageInfo packageInfo = mInstalledList.get(position); 
73            
74           holder.textView.setText( packageInfo.applicationInfo.loadLabel(mInstance.getPackageManager()).toString() ); 
75           holder.imageView.setImageResource(packageInfo.applicationInfo.icon); 
76           Drawable drawable = packageInfo.applicationInfo.loadIcon(mInstance.getPackageManager()); 
77           holder.imageView.setBackgroundDrawable(drawable); 
78            
79           if (contains(mCheckedList, packageInfo)) { 
80               holder.checkBox.setChecked(true); 
81           } else { 
82               holder.checkBox.setChecked(false); 
83           } 
84            
85           convertView.setOnClickListener(new OnClickListener() { 
86    
87               @Override 
88               public void onClick(View view) { 
89    
90                   if (contains(mCheckedList, packageInfo)) { 
91                       remove(mCheckedList, packageInfo); 
92                   } else { 
93                       mCheckedList.add(packageInfo.packageName); 
94                   } 
95                    
96                   BlockUtils.save(mInstance, mCheckedList); 
97                    
98                   notifyDataSetChanged(); 
99               } 
100          }); 
101           
102          return convertView; 
103      }

105      private boolean contains(ArrayList<String> list, PackageInfo item){ 
106          if(list==null || item==null){ 
107              return false; 
108          } 
109           
110          for(int i=0; i<list.size(); i++){ 
111               
112              if(list.get(i).equals(item.packageName)){ 
113                  return true; 
114              } 
115          } 
116           
117          return false; 
118      } 
119       
120      private void remove(ArrayList<String> list, PackageInfo item){ 
121          if(list==null || item==null){ 
122              return; 
123          } 
124           
125          for(int i=0; i<list.size(); i++){ 
126               
127              if(list.get(i).equals(item.packageName)){ 
128                  list.remove(i); 
129                  break; 
130              } 
131          } 
132      }
